AuthMiddleware.java
public class AuthMiddleware implements Middleware {
    @Override
    public void apply(Request req, Response res) 
            throws Exception {
        String token = req.header("Authorization");
        
        if (token == null || !isValid(token)) {
            res.status(401);
            throw new UnauthorizedException();
        }
    }
}

// Apply to controller
@Step(path = "/api/secure")
@Before({AuthMiddleware.class})

    // All endpoints protected ‚úÖ

ErrorHandling.java
@Step(path = "/users/{id}", method = "GET")
public User getUser(Request req) throws Exception {
    String id = req.param("id");
    User user = userService.findById(id);
    
    if (user == null) {
        throw new NotFoundException("User not found");
    }
    
    return user;
}

// Global exception handler
@ExceptionHandler(NotFoundException.class)
public Object handleNotFound(Exception e, Response res) {
    res.status(404);
    return Map.of(
        "error", "Not Found",
        "message", e.getMessage()
    );
}
